15pair<bool, vector<int>> findCycle(vector<vector<int>> &graph, int n) {
16 vector<bool> visited(n, false);
17 vector<bool> inStack(n, false);
18 vector<int> cycle;
19
20 for (int start = 0; start < n; start++) {
21 if (visited[start])
22 continue;
23
24 stack<pair<int, int>> st;
25 vector<int> path;
26 st.push({start, 0});
27 inStack[start] = true;
28 path.push_back(start);
29
30 while (!st.empty()) {
31 int current = st.top().first;
32 int &neighborIndex = st.top().second;
33
34 visited[current] = true;
35
36 bool foundNext = false;
37 for (; neighborIndex < n; neighborIndex++) {
38 if (graph[current][neighborIndex] > 0) {
39 if (!visited[neighborIndex]) {
40 st.push({neighborIndex, 0});
41 inStack[neighborIndex] = true;
42 path.push_back(neighborIndex);
43 foundNext = true;
44 break;
45 } else if (inStack[neighborIndex]) {
46 auto it = find(path.begin(), path.end(), neighborIndex);
47 cycle = vector<int>(it, path.end());
48 return {true, cycle};
49 }
50 }
51 }
52
53 if (!foundNext) {
54 inStack[current] = false;
55 path.pop_back();
56 st.pop();
57 }
58 }
59 }
60 return {false, cycle};
61}
