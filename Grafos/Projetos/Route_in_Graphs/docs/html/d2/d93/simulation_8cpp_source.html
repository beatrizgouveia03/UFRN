127SimulationResult Simulation::initialize(int argc, char *argv[]) {
128 // Inicializa a simulação
129 // Processa os argumentos da linha de comando
130 if (argc <= 1) {
131 return usage("Erro: Nome do arquivo ausente");
132 }
133
134 string filename = argv[1];
135 ifstream file(filename);
136
137 cout << "Inicializando a simulacao..." << endl;
138
139 if (!file.is_open()) {
140 return usage("Erro: Nao foi possivel abrir o arquivo");
141 }
142
143 // Leitura da entrada
144 while (!file.eof()) {
145
146 cout << filename << endl;
147
148 // Read the graph from the file
149 string v1, v2, w;
150 int numNodes;
151
152 string line;
153 char type;
154
155 cout << "Reading number of nodes..." << endl;
156
157 file >> type >> numNodes;
158
159 map<int, string> dictionary;
160 vector<vector<int>> adj = vector<vector<int>>(numNodes, vector<int>(numNodes, -1));
161
162
163 // Read the connections from the file
164 while (getline(file, line)) {
165 if (line == string())
166 continue;
167
168 istringstream ss(line);
169 getline(ss, v1, ' ');
170 getline(ss, v2, ' ');
171 getline(ss, w);
172
173 if(dictionary.empty()){
174 if(v1 == "s"){
175 dictionary[0] = "s";
176 for(int i{1}; i<numNodes-1; ++i){
177 dictionary[i] = string(1, 'a' + (i-1));
178 }
179 dictionary[numNodes-1] = "t";
180 } else {
181 for(int i{0}; i<numNodes; ++i){
182 dictionary[i] = to_string(i+1);
183 }
184 }
185 }
186
187 cout << "Reading connection " << v1 << " " << v2 << " weighting " << w << endl;
188
189 size_t x = INF, y = INF;
190
191 //Update the dictionary
192 for(auto i{0}; i<dictionary.size();++i){
193 if(dictionary[i] == v1){
194 x = size_t(i);
195 }
196 if(dictionary[i] == v2){
197 y = size_t(i);
198 }
199 }
200
201 if(x == INF){
202 x = dictionary.size();
203 dictionary[int(x)] = v1;
204 }
205
206 if(y == INF){
207 y = dictionary.size();
208 dictionary[int(y)] = v2;
209 }
210
211 if (dictionary.size() > numNodes) {
212 return usage("Error: Node out-of-index");
213 }
214
215
216 adj[x][y] = stoi(w);
217
218 (type == 'G') ? adj[y][x] = stoi(w) : -1;
219
220 this->graph.adj = adj;
221 this->graph.n = numNodes;
222 this->graph.dictionary = dictionary;
223 this->graph.type = (type == 'G')? DIRECTED : UNDIRECTED;
224 }
225 }
226
227 cout << "Inicializacao concluida com sucesso" << endl;
228 cout << this->graph << endl;
229
230 return SimulationResult("", simulation_result_e(0));
231}
